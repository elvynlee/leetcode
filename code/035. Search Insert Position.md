    #self write
    class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        l, r = 0, len(nums)
        while l < r:
            m = (l+r) // 2
            if target < nums[m]:
                r = m
            elif target > nums[m]:
                l = m + 1
            else:
                return m
        if l == 0 or l == len(nums) or nums[l] != target: return l
