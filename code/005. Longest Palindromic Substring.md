    def longestPalindrome(s):
        def is_pal(s, l, r):
            while l >= 0 and r < len(s) and s[l] == s[r]:
                l -= 1; r += 1
            return s[l+1:r]
            
        final_str = ''
        for i in range(len(s)):
            #'aba' case
            tmp = is_pal(s, i, i)
            if len(tmp) > len(final_str):
                final_str = tmp
            #'abba' case
            tmp = is_pal(s, i, i+1)
            if len(tmp) > len(final_str):
                final_str = tmp
        return final_str
