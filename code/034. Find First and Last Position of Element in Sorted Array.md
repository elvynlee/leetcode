    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def lr_search(self, nums, target, left):
            l, r = 0, len(nums)
            
            while l < r:
                mid = (l+r) // 2
                if target < nums[mid] or (left and target == nums[mid]):
                    r = mid
                else:
                    l = mid + 1
            return l
        
        li = self.lr_search(nums, target, True)
        
        if li == len(nums) or nums[li] != target:
            return [-1, -1]
        ri = self.lr_search(nums, target, False)-1
        
        return [li, ri]

        l, r = 0, len(nums)-1
        
        while l <= r:
            if target == nums[l]:
                break
            l += 1
        if l > r:
            return [-1, -1]
        while r >= l:
            if target == nums[r]:
                break
            r -= 1
        return [l, r]

    def searchRange(self, nums: List[int], target: int) -> List[int]:
        if nums == []: return [-1, -1]
        l, r = 0, len(nums)
        li, ri = None, None
        while l < r:
            mid = (l + r) // 2
            if target > nums[mid]:
                l = mid + 1
            elif target <= nums[mid]:
                r = mid
        if r >= len(nums) or nums[r] != target: return [-1, -1]
        li = r
        
        l, r = li, len(nums)        
        while l < r:
            mid = (l + r) // 2
            if target >= nums[mid]:
                l = mid + 1
            elif target < nums[mid]:
                r = mid
        ri = l - 1    
                        
        return [li, ri]
